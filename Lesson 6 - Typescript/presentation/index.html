Functions

    return x + y;
}

function funcWithOptionalParameter(s1: string, s2?: string) {
    return s1 + ' ' + s2;
}

function funcWithDefaultParameter(s1: string, s2: string = 'default value') {
    return s1 + ' ' + s2;
}

Function overloading


function func(s1: string, s2: string): string;
function func(s1: number, s2: number): number;
function func(a, b): any {
    if (typeof a === 'string') {
        return a + ' ' + b;
    } else {
        return a + b;
    }
 }

Union types


type StringOrNumber = string | number;

function func(a: StringOrNumber, b: StringOrNumber): StringOrNumber {
    if (typeof a === 'string' && typeof b === 'string') {
        return a + ' ' + b;
    } else if (typeof a === 'number' && typeof b === 'number') {
        return a + b;
    }
 }

Inheritance


class Animal {
    constructor(public name: string) { }
    move(distanceInMeters: number = 0) {
        console.log(`${this.name} moved ${distanceInMeters}m.`);
    }
}

class Snake extends Animal {
    constructor(name: string) { super(name); }
    move(distanceInMeters = 5) {
        console.log("Slithering...");
        super.move(distanceInMeters);
    }
}

class Horse extends Animal {
    constructor(name: string) { super(name); }
    move(distanceInMeters = 45) {
        console.log("Galloping...");
        super.move(distanceInMeters);
    }
}

let sam: Snake = new Snake("Sammy the Python");
let tom: Horse = new Horse("Tommy the Palomino");
let joe: Animal = new Animal("Joe the unknown Animal");
let animalArray: Array<Animal> = [ sam, tom, joe ];

sam.move(6);
tom.move(34);
joe.move(2);
animalArray.forEach(animal => animal.move());
